int dfs(int u ,int father)
{
    int dist = 0 , d1 = 0 ,d2 = 0;
    for(int i = h[u] ; i!= -1 ; i = ne[i]){
        int j = e[i];
        if(j == father)continue;
        int d = dfs(j , u) + w[i];
        dist = max(dist ,d);
        if(d >= d1)d2 = d1 , d1 = d;
        else if(d > d2)d2 = d;
    }
    ans = max(ans , d1 + d2);
    return dist;
}
